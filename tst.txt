from flask import Flask, jsonify

app = Flask(__name__)

@app.route("/api/users", methods=["GET"])
def users():
    return jsonify(
        {
            "users":
            [ 
                'john',
                'doe',
                'jane',
            ]
        }
    )
    
if __name__ == "__name__":
    app.run(debug=True, port=8080)
